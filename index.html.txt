    <script>
        let selectedCourt = null;
        
        function selectCourt(courtNum) {
            selectedCourt = courtNum;
            document.querySelectorAll('.court-btn').forEach(btn => btn.classList.remove('selected'));
            event.target.classList.add('selected');
            document.getElementById('selected-court').innerText = `专砖 ${courtNum} 专`;
        }
        
        function bookCourt() {
            const date = document.getElementById('date').value;
            const time = document.getElementById('time').value;
            
            if (!selectedCourt) {
                alert(' 专 专砖 拽!');
                return;
            }
            if (!date || !time) {
                alert('  转专 砖注!');
                return;
            }
            
            alert(` 砖专! 专砖 ${selectedCourt} 转专 ${date} 砖注 ${time}. 转! `);
        }
        
        // 专 转专  专专转 
        document.getElementById('date').valueAsDate = new Date();
    </script>
